34    int A[N],B[N],C[N];
35    int rev[N],pre[N];
36    int init(int n){
37        int lim=0;
38        while((1<<lim)<=n) lim++;
39        for(int i=0;i<(1<<lim);i++)
40            rev[i]=(rev[i>>1]>>1) | ((i & 1)<<(lim-1));
41        
42        int omega=ksmmod(3,(mod-1)/(1<<lim));
43        pre[0]=1;
44        for(int i=1;i<=(1<<lim);i++)
45            pre[i]=pre[i-1]*omega%mod;
46        return lim;

48    void ntt(int *f,int n,int opt){
49        for(int i=0;i<n;i++)
50            if(i<rev[i]) swap(f[i],f[rev[i]]);
51        for(int l=1;l<n;l*=2){
52            for(int i=0;i<n;i+=l*2){
53                int u=n/l/2,nw=(opt==1)?0:n;
54                for(int j=0;j<l;j++){
55                    int x=f[i+j],y=pre[nw]*f[i+j+l]%mod;
56                    gmod(f[i+j]=x+y-mod),gmod(f[i+j+l]=x-y);
57                    nw+=opt*u;
58                }
59            }
60        }
61        if(opt==-1){
62            int t=ksmmod(n,mod-2);
63            for(int i=0;i<=n-1;i++)
64                f[i]=f[i]*t%mod;
65        }
66    }

79        for(int i=0;i<=n;i++) A[i]=f[i];
80        for(int i=0;i<=m;i++) B[i]=g[i];
81        ntt(A,1<<lim,1);

84        ntt(C,1<<lim,-1);
