34    void ntt(int *f,int n,int opt){
35        for(int i=0;i<n;i++)
36            if(i<rev[i]) swap(f[i],f[rev[i]]);
37        for(int l=1;l<n;l*=2){
38            for(int i=0;i<n;i+=l*2){
39                int u=n/l/2,nw=(opt==1)?0:n;
40                for(int j=0;j<l;j++){

42                    gmod(f[i+j]=x+y-mod),gmod(f[i+j+l]=x-y);
43                    nw+=opt*u;
44                }
