using namespace std;
const int N=2e5+5,INF=1e9;
int T,n,m,A,B,ans;
vector <int> G[N];
int L[N],V[N];


int dis[N];
queue <int> q;

int solve(int st,int ed){
    for(int i=1;i<=n;i++) dis[i]=INF;
    while(q.size()) q.pop();
    
    dis[st]=0;
    q.push(st);
    while(q.size()){
        int u=q.front(); q.pop();
        if(dis[u]>V[u]) continue;
        
        for(int v: G[u]){
            if(dis[v]==INF){
                dis[v]=dis[u]+1;
                q.push(v);
            }
        }
    }
    if(dis[ed]>V[ed]) return INF;
    else return dis[ed];
}

bool check(int k){
    for(int i=1;i<=n;i++) V[i]=k-V[i];
    int res=solve(n,1);
    for(int i=1;i<=n;i++) V[i]=k-V[i];
    return (res<INF);
}


struct abc{
    int x;
};
bool operator <(abc X,abc Y){
    return V[X.x]>V[Y.x];
}
priority_queue <abc> Q;
bool vis[N];
bool chk(){
    for(int i=1;i<=n;i++) vis[i]=0;
    while(Q.size()) Q.pop();
    
    int cur=0;
    Q.push({1});
    vis[1]=1;
    while(Q.size()){
        int u=Q.top().x; Q.pop();
        if(V[u]>cur) return 0;
        if(u==n) return 1;
        for(int v: G[u]){
            if(!vis[v]) Q.push({v}),vis[v]=1;
        }
        cur++;
    }
    return 0;
}
