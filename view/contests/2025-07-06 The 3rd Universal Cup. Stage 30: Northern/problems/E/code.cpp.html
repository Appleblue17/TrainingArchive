37inline ll calc(int d, int sx, int sy, int ex, int ey){

40    if(h[d].find(state)!=h[d].end()) return h[d][state];
41    int id1=((sx>(1<<(d-1)))?2:0)+((sy>(1<<(d-1)))?1:0);
42    int id2=((ex>(1<<(d-1)))?2:0)+((ey>(1<<(d-1)))?1:0);
43    if(sx>(1<<(d-1))) sx-=(1<<(d-1));
