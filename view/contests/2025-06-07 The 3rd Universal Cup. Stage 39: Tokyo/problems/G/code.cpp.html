50vector<Point> ConvexHull(vector<Point>& p) {
51  // 预处理，删除重复点
52  sort(p.begin(), p.end());
53  p.erase(unique(p.begin(), p.end()), p.end());
54
55  int n = p.size();
56  int m = 0;
57  vector<Point> ch(n+1);
58  for(int i = 0; i < n; i++) {
59    while(m > 1 && Cross(ch[m-1]-ch[m-2], p[i]-ch[m-2]) <= 0) m--;
60    ch[m++] = p[i];
61  }
62  int k = m;
63  for(int i = n-2; i >= 0; i--) {
64    while(m > k && Cross(ch[m-1]-ch[m-2], p[i]-ch[m-2]) <= 0) m--;
65    ch[m++] = p[i];
66  }
67  if(n > 1) m--;
68  ch.resize(m);
69  return ch;
70}
