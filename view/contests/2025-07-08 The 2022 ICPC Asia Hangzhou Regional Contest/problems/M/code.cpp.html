18void dfs(int x, int fa){
19    for(auto edg:e[x]){
20        int y=edg.fi, z=edg.se;
21        if(y==fa) continue;
22        dfs(y, x);
23        sum[x]+=sum[y]+1ll*sz[y]*z;
24        if(!sz[y]) continue;
25        if(sz[x]){
26            d[x].se=gcd(gcd(abs(d[y].fi+z-d[x].fi), d[y].se), d[x].se); 
27        }
28        else{
29            d[x].se=gcd(d[y].se, d[x].se); 
30        }

32        d[x].fi=d[y].fi+z;
33    }
